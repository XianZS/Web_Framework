# 3、约数之和

```python
a,b=map(int,input().split())
mod=9901
ans=1

def exp(num,k):
    if num==0:
        return 0
    if num==1 or k==0:
        return 1
    res=exp(num,k//2)
    if k&1:
        return res*res*num%mod
    else:
        return res*res%mod

def make(num,k):
    if k==0:
        return 1
    if k&1==0: # 奇数
        return (num%mod*make(num,k-1)+1)%mod
    # 偶数
    return (1+exp(num,k//2+1))*make(num,k//2)%mod

i=2
while i<=a:
    s=0
    while a%i==0:
        s+=1
        a=a//i
    if s:
        ans=ans*make(i,s*b)%mod
    i+=1
if a==0:
    print(0)
else:
    print(ans)
```
